# ::snt Who composed the soundtrack for Cameron's Titanic?
(c / compose-02
      :ARG0 (a / amr-unknown)
      :ARG1 (s / soundtrack
            :poss (s2 / ship
                  :name (n / name
                        :op1 "Titanic")
                  :poss (p / person
                        :name (n2 / name
                              :op1 "Cameron")))))
