# ::snt Who is starring in Spanish movies produced by Benicio del Toro?
(s / star-01
      :ARG1 (a / amr-unknown)
      :ARG2 (m / movie
            :mod (c / country
                  :name (n / name
                        :op1 "Spain"))
            :ARG1-of (p / produce-01
                  :ARG0 (p2 / person
                        :name (n2 / name
                              :op1 "Benicio"
                              :op2 "del"
                              :op3 "Torchio")))))
