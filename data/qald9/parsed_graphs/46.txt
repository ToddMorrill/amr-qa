# ::snt Show a list of soccer clubs that play in the Bundesliga.
(s / show-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (l / list
            :consist-of (c / club
                  :mod (s2 / soccer)
                  :ARG0-of (p / play-01
                        :ARG3 (l2 / league
                              :name (n / name
                                    :op1 "Bundesliga"))))))
