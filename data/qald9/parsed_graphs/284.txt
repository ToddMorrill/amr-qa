# ::snt What is the highest mountain in Italy?
(m / mountain
      :location (c / country
            :name (n / name
                  :op1 "Italy"))
      :ARG1-of (h / have-degree-91
            :ARG2 (h2 / high-02
                  :ARG1 m)
            :ARG3 (m2 / most))
      :domain (a / amr-unknown))
