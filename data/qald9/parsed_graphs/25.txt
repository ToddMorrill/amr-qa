# ::snt Which U.S. states are in the same timezone as Utah?
(b / be-located-at-91
      :ARG1 (s / state
            :mod (a / amr-unknown)
            :location (c / country
                  :name (n / name
                        :op1 "U.S.")))
      :ARG2 (z / zone
            :mod (t / time)
            :ARG1-of (s2 / same-01
                  :ARG2 (s3 / state
                        :name (n2 / name
                              :op1 "Utah")))))
