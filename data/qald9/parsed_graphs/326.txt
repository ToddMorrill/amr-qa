# ::snt List all episodes of the first season of the HBO television series The Sopranos.
(l / list-01
      :ARG1 (e / episode
            :mod (a / all)
            :part-of (s / season
                  :ord (o / ordinal-entity
                        :value 1)
                  :part-of (b / broadcast-program
                        :name (n / name
                              :op1 "The"
                              :op2 "Sopranos")
                        :mod (t / television)
                        :medium (c / company
                              :name (n2 / name
                                    :op1 "HBO"))))))
