      :ARG1 (a / amr-unknown)
      :ARG3 (a2 / and

                  :name (n / name
                        :op1 "Manhattan"
                        :op2 "Project"))
            :op2 (a3 / award
                  :name (n2 / name
                        :op1 "Nobel"
                        :op2 "Peace"
                        :op3 "Prize"))))
