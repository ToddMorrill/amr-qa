# ::snt When will start the final match of the football world cup 2018?
(s / start-01
      :ARG1 (m / match-03
            :mod (f / final)
            :subevent-of (g / game
                  :name (n / name
                        :op1 "Football"
                        :op2 "World"
                        :op3 "Cup")
                  :time (d / date-entity
                        :year 2018)))
      :time (a / amr-unknown))
