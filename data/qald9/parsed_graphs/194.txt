# ::snt What is the alma mater of the chancellor of Germany Angela Merkel?
(a / alma-mater
      :domain (a2 / amr-unknown)
      :poss (p / person
            :name (n / name
                  :op1 "Angela"
                  :op2 "Merkel")
            :ARG0-of (h / have-org-role-91
                  :ARG1 (c / country
                        :name (n2 / name
                              :op1 "Germany"))
                  :ARG2 (c2 / chancellor))))
