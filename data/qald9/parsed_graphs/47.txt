# ::snt Which volcanos in Japan erupted since 2000?
(e / erupt-01
      :ARG1 (v / volcano
            :mod (a / amr-unknown)
            :location (c / country
                  :name (n / name
                        :op1 "Japan")))
      :time (s / since
            :op1 (d / date-entity
                  :year 2000)))
