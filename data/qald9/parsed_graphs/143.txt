# ::snt Give me a list of all lakes in Denmark.
(g / give-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (l / list
            :consist-of (l2 / lake
                  :mod (a / all)
                  :location (c / country
                        :name (n / name
                              :op1 "Denmark"))))
      :ARG2 (ii / i))
