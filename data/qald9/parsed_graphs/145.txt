# ::snt Show me all English Gothic buildings in Kent.
(s / show-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (b / building
            :mod (g / Gothic)
            :mod (c / country
                  :name (n / name
                        :op1 "England"))
            :mod (a / all)
            :location (c2 / county
                  :name (n2 / name
                        :op1 "Kent")))
      :ARG2 (ii / i))
