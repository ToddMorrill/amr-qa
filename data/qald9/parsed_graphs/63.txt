# ::snt In which films directed by Garry Marshall was Julia Roberts starring?
(s / star-01
      :ARG1 (p / person
            :name (n / name
                  :op1 "Julia"
                  :op2 "Roberts"))
      :ARG2 (f / film
            :ARG1-of (d / direct-01
                  :ARG0 (p2 / person
                        :name (n2 / name
                              :op1 "Garry"
                              :op2 "Marshall")))
            :mod (a / amr-unknown)))
