# ::snt What is the total population of Melbourne, Florida?
(p / population
      :quant (a / amr-unknown)
      :ARG2-of (t / total-01)
      :poss (c / city
            :name (n / name
                  :op1 "Melbourne")
            :location (s / state
                  :name (n2 / name
                        :op1 "Florida"))))
