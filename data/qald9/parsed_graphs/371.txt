# ::snt Give me all books by William Goldman with more than 300 pages.
(g / give-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (b / book
            :mod (a / all)
            :source (p / person
                  :name (n / name
                        :op1 "William"
                        :op2 "Goldman"))
            :ARG1-of (p2 / page-01
                  :quant (m / more-than
                        :op1 300)))
      :ARG2 (ii / i))
