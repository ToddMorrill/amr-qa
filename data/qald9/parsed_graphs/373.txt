# ::snt Which books by Kerouac were published by Viking Press?
(p / publish-01
      :ARG0 (c / company
            :name (n / name
                  :op1 "Viking"
                  :op2 "Press"))
      :ARG1 (b / book
            :mod (a / amr-unknown)
            :source (p2 / person
                  :name (n2 / name
                        :op1 "Kerouac"))))
