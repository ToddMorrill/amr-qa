      :ARG0 (p / person
            :quant (a / amr-unknown))
      :ARG1 (s / stuff
            :ARG1-of (b / broadcast-01
                  :ARG0 (p2 / publication
                        :name (n / name
                              :op1 "Tokyo"
                              :op2 "Broadcasting"
                              :op3 "System")))))
