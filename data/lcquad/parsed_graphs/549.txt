# ::snt Name the line of the Vadodara Junction railway station which is also the  hub airport of the Pawan Hans?
(n / name-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (l / line
            :poss (s / station
                  :name (n2 / name
                        :op1 "Vadodara"
                        :op2 "Junction")
                  :mod (r / railway)
                  :mod (h / hub
                        :mod (a / also)
                        :poss (a2 / airport
                              :name (n3 / name
                                    :op1 "Pawan"
                                    :op2 "Hans"))))))
