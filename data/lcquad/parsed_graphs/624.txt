# ::snt What is the television show whose company is Playtone and written by Erik Jendresen?
(s / show-04
      :ARG1 (a / amr-unknown)
      :ARG3 (t / television)
      :poss (c / company
            :name (n / name
                  :op1 "Playtone"))
      :ARG1-of (w / write-01
            :ARG0 (p / person
                  :name (n2 / name
                        :op1 "Erik"
                        :op2 "Jendresen"))))
