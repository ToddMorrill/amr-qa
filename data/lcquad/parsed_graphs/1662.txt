      :ARG2 (r / region
            :mod (o / other)
            :quant (a / amr-unknown))

            :ARG0-of (s2 / serve-01
                  :ARG2 (o2 / ocean
                        :name (n / name
                              :op1 "Pacific"
                              :op2 "Ocean")))))
