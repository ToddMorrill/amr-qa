# ::snt List bacteria whose order (taxonomy) is Bacillales and domain is Bacteria?
(l / list-01
      :ARG1 (b / bacterium
            :mod (o / order
                  :name (n / name
                        :op1 "Bacillales"))
            :mod (t / taxonomy)
            :mod (d / domain
                  :name (n2 / name
                        :op1 "Bacteria"))))
