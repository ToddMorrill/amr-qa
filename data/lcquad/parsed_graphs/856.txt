# ::snt How many railway lines go through the station maintained by Public transport authority of western australia?
(g / go-01
      :ARG1 (l / line
            :mod (r / railway)
            :quant (a / amr-unknown))
      :path (s / station
            :ARG1-of (m / maintain-01
                  :ARG0 (a2 / authority
                        :mod (t / transport-01
                              :ARG1-of (p / public-02))
                        :poss (w / west
                              :part-of (c / country
                                    :name (n / name
                                          :op1 "Australia")))))))
