# ::snt What are the movies whose director's daughter is Luke Scott?
(m / movie
      :domain (a / amr-unknown)
      :poss (p / person
            :name (n / name
                  :op1 "Luke"
                  :op2 "Scott")
            :ARG0-of (h / have-rel-role-91
                  :ARG1 (p2 / person
                        :ARG0-of (d / direct-01))
                  :ARG2 (d2 / daughter))))
