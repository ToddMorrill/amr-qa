# ::snt Where does Dennis DeConcini and Scott Berkun both live?
(l / live-01
      :ARG0 (a / and
            :op1 (p / person
                  :name (n / name
                        :op1 "Dennis"
                        :op2 "DeConcini"))
            :op2 (p2 / person
                  :name (n2 / name
                        :op1 "Scott"
                        :op2 "Berkun"))
            :mod (b / both))
      :location (a2 / amr-unknown))
