# ::snt What is the subject of the book which preceded Sagan's Cosmos?
(s / subject
      :domain (a / amr-unknown)
      :topic-of (b / book
            :ARG1-of (p / precede-01
                  :name (n / name
                        :op1 "Cosmos")
                  :poss (p2 / person
                        :name (n2 / name
                              :op1 "Sagan")))
            :ARG2-of p))
