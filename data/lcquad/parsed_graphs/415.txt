# ::snt Name the sitcoms whose network's owning company is Chris-Craft Industries?
(n / name-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (s / sitcom
            :poss (c / company
                  :name (n2 / name
                        :op1 "Chris-Craft"
                        :op2 "Industries")
                  :ARG0-of (o / own-01
                        :ARG1 (n3 / network
                              :poss c)))))
