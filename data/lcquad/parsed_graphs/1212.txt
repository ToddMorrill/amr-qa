# ::snt What are some games availible on nintendo's virtual console?
(a / available-02
      :ARG2 (g / game
            :quant (s / some)
            :mod (a2 / amr-unknown))
      :location (c / console
            :mod (v / virtual)
            :poss (c2 / company
                  :name (n / name
                        :op1 "Nintendo"))))
