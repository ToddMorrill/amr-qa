# ::snt How many awards have graduates from the University of Queensland earned?
(e / earn-01
      :ARG0 (p / person
            :ARG1-of (g / graduate-01
                  :ARG2 (u / university
                        :name (n / name
                              :op1 "University"
                              :op2 "of"
                              :op3 "Queensland"))))
      :ARG1 (a / award-01
            :ARG2 p
            :quant (a2 / amr-unknown)))
