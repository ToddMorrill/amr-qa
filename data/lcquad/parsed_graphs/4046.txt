      :ARG0 (a / and
            :op1 (r / road
                  :name (n / name
                        :op1 "Jerome"

                  :name (n2 / name
                        :op1 "Tremont"
                        :op2 "Avenue")))
      :ARG1 (c / county
            :mod (a2 / amr-unknown)))
