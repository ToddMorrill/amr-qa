# ::snt What are the television shows whose company's predecessor is Desilu Productions?
(s / show-04
      :ARG1 (a / amr-unknown)
      :ARG3 (t / television)
      :poss (c / company
            :name (n / name
                  :op1 "Desilu"
                  :op2 "Productions")
            :ARG1-of (p / precede-01
                  :ARG2 (c2 / company))))
