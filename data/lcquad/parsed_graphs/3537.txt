# ::snt Where is the fictional character from, whose last appearance was in the Avengers, 1998?
(c / character
      :source (a / amr-unknown)
      :mod (f / fiction)
      :ARG1-of (a2 / appear-01
            :ARG2 (p / publication
                  :name (n / name
                        :op1 "Avengers"))
            :time (d / date-entity
                  :year 1998)
            :mod (l / last)))
