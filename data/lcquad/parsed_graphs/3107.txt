# ::snt Which politician was succeeded by someone who graduated from the Instituts d'tudes politiques?
(s / succeed-02
      :ARG0 (s2 / someone
            :ARG1-of (g / graduate-01
                  :ARG2 (s3 / school
                        :name (n / name
                              :op1 "Instituts"
                              :op2 "d'tudes"
                              :op3 "politiques"))))
      :ARG1 (p / politician
            :mod (a / amr-unknown)))
