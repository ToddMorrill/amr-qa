      :domain (a2 / amr-unknown)
      :part-of (a3 / album
            :ARG0-of (h / have-03
                  :ARG1 (s / song
                        :name (n / name
                              :op1 "I"
                              :op2 "Can't"
                              :op3 "Change"
                              :op4 "the"
                              :op5 "World")))))
