# ::snt In how many nations does Rhaetian Railways own train stations?
(o / own-01
      :ARG0 (c / company
            :name (n / name
                  :op1 "Rhaetian"
                  :op2 "Railways"))
      :ARG1 (s / station
            :mod (t / train))
      :location (n2 / nation
            :quant (a / amr-unknown)))
