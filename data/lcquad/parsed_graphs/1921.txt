# ::snt How many people have managed a club which owns the Riverside Stadium?
(m / manage-01
      :ARG0 (p / person
            :quant (a / amr-unknown))
      :ARG1 (c / club
            :ARG0-of (o / own-01
                  :ARG1 (s / sports-facility
                        :name (n / name
                              :op1 "Riverside"
                              :op2 "Stadium")))))
