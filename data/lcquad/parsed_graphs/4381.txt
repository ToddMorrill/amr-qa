# ::snt Name the university whose athletic department is called National Collegiate Athletic Association and has a chancellor named Nicholas S. Zeppos?
(n / name-01
      :mode imperative
      :ARG0 (y / you)
      :ARG1 (u / university
            :poss-of (d / department
                  :mod (a / athletic)
                  :ARG1-of (c / call-01
                        :ARG2 (n2 / name
                              :op1 "National"
                              :op2 "Colcoliate"
                              :op3 "Athletic"
                              :op4 "Association")))
            :ARG0-of (h / have-03
                  :ARG1 (c2 / chancellor
                        :ARG1-of n
                        :ARG2 (p / person
                              :name (n3 / name
                                    :op1 "Nicholas"
                                    :op2 "S."
                                    :op3 "Zeppos"))))))
