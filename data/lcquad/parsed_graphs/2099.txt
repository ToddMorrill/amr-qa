# ::snt What is the region of the ethnic group which speaks the language of Arkansas?
(r / region
      :domain (a / amr-unknown)
      :poss (e / ethnic-group
            :ARG0-of (s / speak-01
                  :ARG3 (l / language
                        :poss (s2 / state
                              :name (n / name
                                    :op1 "Arkansas"))))))
