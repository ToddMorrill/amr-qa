# ::snt Who was once married to both Kelly Brook and Rosie Huntington-Whiteley?
(m / marry-01
      :ARG1 (a / amr-unknown)
      :ARG2 (a2 / and
            :op1 (p / person
                  :name (n / name
                        :op1 "Kelly"
                        :op2 "Brook"))
            :op2 (p2 / person
                  :name (n2 / name
                        :op1 "Rosie"
                        :op2 "Huntington-Whiteley"))
            :mod (b / both))
      :time (o / once))
