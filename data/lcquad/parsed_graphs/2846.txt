# ::snt Which line of the Vadodara Junction railway station is the route end of Mumbai Vadodara Expressway ?
(e / end-01
      :ARG1 (r / route
            :poss (e2 / expressway
                  :name (n / name
                        :op1 "Mumbai"
                        :op2 "Vadodara"
                        :op3 "Expressway")))
      :ARG2 (l / line
            :mod (a / amr-unknown)
            :part-of (s / station
                  :name (n2 / name
                        :op1 "Vadodara"
                        :op2 "Junction")
                  :mod (r2 / railway))))
