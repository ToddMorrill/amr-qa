# ::snt Which band's  former member are Kevin Jonas and Joe Jonas?
(m / member
      :time (f / former)
      :poss (b / band
            :mod (a / amr-unknown))
      :domain (a2 / and
            :op1 (p / person
                  :name (n / name
                        :op1 "Kevin"
                        :op2 "Jonas"))
            :op2 (p2 / person
                  :name (n2 / name
                        :op1 "Joe"
                        :op2 "Jonas"))))
