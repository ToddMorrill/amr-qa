# ::snt Which TV show produced by Paramount Television and is located in Massachusetts ?
(s / show-04
      :ARG3 (t / television)
      :ARG1-of (p / produce-01
            :ARG0 (c / company
                  :name (n / name
                        :op1 "Paramount"
                        :op2 "Television")))
      :location (s2 / state
            :name (n2 / name
                  :op1 "Massachusetts"))
      :mod (a / amr-unknown))
