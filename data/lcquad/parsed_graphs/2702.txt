# ::snt Which country has the Carmel Winery and the Teperberg 1870 winery ?
(h / have-03
      :ARG0 (c / country
            :mod (a / amr-unknown))
      :ARG1 (a2 / and
            :op1 (w / winery
                  :name (n / name
                        :op1 "Carmel"
                        :op2 "Winery"))
            :op2 (w2 / winery
                  :name (n2 / name
                        :op1 "Teperberg"
                        :op2 1870))))
