# ::snt To which county of New York city does the Tremount Avenue street belong?
(b / belong-01
      :ARG0 (s / street
            :name (n / name
                  :op1 "Tremount"
                  :op2 "Avenue"))
      :ARG1 (c / county
            :part-of (c2 / city
                  :name (n2 / name
                        :op1 "New"
                        :op2 "York"))
            :mod (a / amr-unknown)))
