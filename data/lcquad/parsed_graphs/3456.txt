# ::snt How many software run on operating system is Microsoft Windows?
(r / run-01
      :ARG1 (s / software
            :quant (a / amr-unknown))
      :location (s2 / system
            :ARG0-of (o / operate-01))
      :domain (p / product
            :name (n / name
                  :op1 "Windows")
            :poss (c / company
                  :name n
                  :op1 "Microsoft")))
