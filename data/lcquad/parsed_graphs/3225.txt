# ::snt What are the television shows whose distributor is a division of Warner Bros. International TV?
(s / show-04
      :ARG1 (a / amr-unknown)
      :mod (t / television)
      :ARG0-of (d / distribute-01
            :ARG2 (d2 / division
                  :poss (c / company
                        :name (n / name
                              :op1 "Warner"
                              :op2 "Bros."
                              :op3 "International"
                              :op4 "TV")))))
