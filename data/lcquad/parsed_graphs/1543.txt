# ::snt What is the former team of the american football players whose position is Running back?
(t / team
      :domain (a / amr-unknown)
      :time (f / former)
      :poss (p / person
            :ARG0-of (p2 / play-01
                  :ARG1 (f2 / football))
            :mod (c / country
                  :name (n / name
                        :op1 "America"))
            :ARG1-of (p3 / position-01
                  :ARG2 (r / running-back))))
