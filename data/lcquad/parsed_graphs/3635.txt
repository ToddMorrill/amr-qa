# ::snt Who was the coach of 1972 Oakland Raiders season ?
(p / person
      :ARG0-of (c / coach-01
            :ARG1 (s / season
                  :mod (t / team
                        :name (n / name
                              :op1 "Oakland"
                              :op2 "Raiders"))
                  :time (d / date-entity
                        :year 1972)))
      :domain (a / amr-unknown))
