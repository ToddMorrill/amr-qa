# ::snt Which title of Zaman Shah Durrani is also the nationality of Salahuddin ?
(n / nationality
      :poss (p / person
            :name (n2 / name
                  :op1 "Salahuddin"))
      :domain (t / title
            :poss (p2 / person
                  :name (n3 / name
                        :op1 "Zaman"
                        :op2 "Shah"
                        :op3 "Durani"))
            :mod (a / amr-unknown))
      :mod (a2 / also))
