# ::snt Which HBO's television series was produced by Erik Bork?
(p / produce-01
      :ARG0 (p2 / person
            :name (n / name
                  :op1 "Erik"
                  :op2 "Bork"))
      :ARG1 (s / series
            :polarity (a / amr-unknown)
            :mod (t / television)
            :poss (n2 / network
                  :name (n3 / name
                        :op1 "HBO"))))
