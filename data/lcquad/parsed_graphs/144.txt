# ::snt In how many nations can I find dishes made of Shallot?
(f / find-01
      :ARG0 (ii / i)
      :ARG1 (d / dish
            :ARG1-of (m / make-01
                  :ARG2 (p / product
                        :name (n / name
                              :op1 "Shallot"))))
      :location (n2 / nation
            :quant (a / amr-unknown))
      :ARG1-of (p2 / possible-01))
