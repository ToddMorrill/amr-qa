# ::snt What railway lines go through the stations maintained by Western Australian Public Transport Authority?
(g / go-01
      :ARG1 (l / line
            :mod (r / railway)
            :mod (a / amr-unknown))
      :path (s / station
            :ARG1-of (m / maintain-01
                  :ARG0 (g2 / government-organization
                        :name (n / name
                              :op1 "Western"
                              :op2 "Australian"
                              :op3 "Public"
                              :op4 "Transport"
                              :op5 "Authority")))))
